//PARAGRAFOS E ESCRITINHAS
    <hr/> --> faz uma linha bonitinha
    <br> --> pula linha
    <p></p> --> faz paragrafo show
    
    <b></b> --> negrito
    <i></i> --> italico
    <u></u> --> sublinhado
    <s></s> --> riscado
    <center></center> --> centraliza
    <marque></marque> --> texto fica passeando
    
    <title></title> --> faz um titulo
    <h1></h1> --> faz um header, pode ir de 1 a 6 (quanto menor o nº, maior o header)

    <div align=center></div> --> alinha texto no centro 
    <font face=FONTE size=TAMANHO color=COR></font> --> frufruzinhos pro texto


//LISTAS
    <ul></ul> --> separa em uma lista de bolinhas
    <ol></ol> --> separa lista de numeros
        <li></li> --> topicos da lista

    <dl></d1> --> inicia lista de definição
        <dt> termo
            <dd> definição

//IMAGENS E LINKS
    <body background=LINK DA IMAGEM> --> deixa uma imagem de background

    <a href=LINK>TEXTO</a> --> link no texto
